## MHA

```python
import torch.nn as nn
import torch.nn.functional as F
from rotary_embedding_torch import RotaryEmbedding

class MHA(nn.Module):
  def __init__(self, num_head, hidden_dim):
    super().__init__()
    self.num_head = num_head
    self.hidden_dim = hidden_dim
    self.head_dim = hidden_dim // num_head

    self.q_proj = nn.Linear(hidden_dim, hidden_dim, bias=False)
    self.k_proj = nn.Linear(hidden_dim, hidden_dim, bias=False)
    self.v_proj = nn.Linear(hidden_dim, hidden_dim, bias=False)
    self.out_proj = nn.Linear(hidden_dim, hidden_dim, bias=False)

    self.rope = RotaryEmbedding(self.head_dim)

  def forward(self, x, mask=None):
    B, T, _ = x.shape
    q = self.q_proj(x)
    k = self.k_proj(x)
    v = self.v_proj(x)
    # reshape
    q = q.view(B, T, self.num_head, self.head_dim).transpose(1, 2)
    k = k.view(B, T, self.num_head, self.head_dim).transpose(1, 2)
    v = v.view(B, T, self.num_head, self.head_dim).transpose(1, 2)

    # rope
    q = self.rope.rotate_queries_or_keys(q)
    k = self.rope.rotate_queries_or_keys(k)
    # Q@K^T/sqrt(d) = S
    P = q @ k.transpose(-1, -2) / (self.head_dim ** 0.5)
    # mask
    if mask is not None:
      P = P.masked_fill(mask == 0, float('-inf'))
    # softmax
    S = F.softmax(P, dim=-1)
    # @V
    out = S @ v
    # merge head
    out = out.transpose(1, 2).contiguous().view(B, T, -1)

    return self.out_proj(out)
```



## MoE

> MoE = 复用 FFN（expert）+ router + dispatch

```python
class MoE(nn.Module):
  def __init__(self, dim, num_experts, top_k):
    super().__init__()
    self.top_k = top_k
    self.gate = nn.Linear(dim, num_experts, bias=False)
    self.experts = nn.ModuleList([FFN(dim) for _ in range(num_experts)])  # 复用下面的 FFN

  def forward(self, x):
    B, T, d = x.shape
    x = x.view(-1, d)                                     # [N, d] 摊平

    # --- router: 返回 topk 权重 + idx ---
    logits = self.gate(x)                                 # [N, E]
    w, idx = logits.softmax(-1).topk(self.top_k, dim=-1)  # [N, k] 权重 + 选中的专家
    w = w / w.sum(-1, keepdim=True)                       # top-k 内重归一化

    # --- dispatch: per expert ---
    out = torch.zeros_like(x)
    for e, expert in enumerate(self.experts):                  # 循环专家 (E 次), 不循环 token
      token_idx, token_pos = (idx == e).nonzero(as_tuple=True) # 哪个 token / 它的第几个 top-k 选择
      if len(token_idx) == 0:
        continue
      # x[token_idx]		"=" 右边：选出来的密集子集拷贝
      # out[token_idx]	"=" 左边：稀疏地修改若干行
      out[token_idx] += w[token_idx, token_pos].view(-1, 1) * expert(x[token_idx])
    return out.view(B, T, d), logits                           # logits 带出去算 aux loss
```

追问再补 FFN（Llama SwiGLU 版，可直接 call 不展开）：

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class FFN(nn.Module):
  def __init__(self, dim, hidden_dim=None):
    super().__init__()
    hidden_dim = hidden_dim or dim // 2
    self.gate_proj = nn.Linear(dim, hidden_dim, bias=False)
    self.up_proj   = nn.Linear(dim, hidden_dim, bias=False)
    self.down_proj = nn.Linear(hidden_dim, dim, bias=False)

  def forward(self, x):
    return self.down_proj(F.silu(self.gate_proj(x)) * self.up_proj(x))
```
