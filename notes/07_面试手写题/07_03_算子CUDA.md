## Matmul - Tiled

- allocate SMEM
- position in C
- for each tile
  - define SMEM → get idx
  - load A → load B → (sync) → accumulate → sync
- store

```cpp
// [M, K] @ [K, N] = [M, N]
#define CEIL(a, b) ((a + b - 1) / b)
#define BLOCKSIZE 32

__global__ void matMulKernel(const float *A, const float *B, float *C, int M, int N, int K) {
  // assign SMEM
  __shared__ float As[BLOCKSIZE][BLOCKSIZE + 1];
  __shared__ float Bs[BLOCKSIZE][BLOCKSIZE + 1];
  int row = blockIdx.y * blockDim.y + threadIdx.y;
  int col = blockIdx.x * blockDim.x + threadIdx.x;
  // loop each tile
  float sum = 0;
  int numTiles = CEIL(K, BLOCKSIZE);
  for (int t = 0; t < numTiles; t++) {
    int offset = t * BLOCKSIZE;
    // load A
    int col_A = offset + threadIdx.x;
    if (row < M && col_A < K) As[threadIdx.y][threadIdx.x] = A[row * K + col_A];
    else As[threadIdx.y][threadIdx.x] = 0;
    // load B
    int row_B = offset + threadIdx.y;
    if (row_B < K && col < N) Bs[threadIdx.y][threadIdx.x] = B[row_B * N + col];
    else Bs[threadIdx.y][threadIdx.x] = 0;
    // sync
    __syncthreads();
    // accumulate
    for (int k = 0; k < BLOCKSIZE; k++) {
      sum += As[threadIdx.y][k] * Bs[k][threadIdx.x];
    }
    __syncthreads();
  }
  // store
  if (row < M && col < N) C[row * N + col] = sum;
}
```

```cpp
void matMul(const float *A, const float *B, float *C, int M, int N, int K) {
  dim3 threadPerBlock(BLOCKSIZE, BLOCKSIZE);
  dim3 blockPerGrid(CEIL(N, BLOCKSIZE), CEIL(M, BLOCKSIZE));
  matMulKernel<<<blockPerGrid, threadPerBlock>>>(A, B, C, M, N, K);
  cudaDeviceSynchronize();
}
```



## Reduce

- define idx, thread_idx, warp_idx, lane_idx
- warp shuffle 1 → SMEM → warp 0 shuffle → write

```cpp
#define BLOCKSIZE 1024
#define WARPSIZE 32

__global__ void reduceKernel(const float *A, float *out, int N) {
  int idx = threadIdx.x + blockIdx.x * blockDim.x;
  int tid = threadIdx.x;
  int warp_id = tid / WARPSIZE, lane_id = tid % WARPSIZE;
  __shared__ float warp_sum[BLOCKSIZE / WARPSIZE];
  // warp reduce 1
  float val = (idx < N) ? A[idx] : 0;
  for (int offset = WARPSIZE >> 1; offset > 0; offset >>= 1) {
    val += __shfl_down_sync(0xFFFFFFFF, val, offset);
  }
  // save → SMEM (lane 0)
  if (lane_id == 0) warp_sum[warp_id] = val;
  __syncthreads();
  // warp reduce 2
  if (warp_id == 0) {
    float block_val = (lane_id < BLOCKSIZE / WARPSIZE) ? warp_sum[lane_id] : 0;
    for (int offset = WARPSIZE >> 1; offset > 0; offset >>= 1) {
      block_val += __shfl_down_sync(0xFFFFFFFF, block_val, offset);
    }
    // write result
    if (lane_id == 0) atomicAdd(out, block_val);
  }
}
```

```cpp
void reduce(const float *A, float *out, int N) {
  dim3 threadPerBlock(BLOCKSIZE);
  dim3 blockPerGrid(CEIL(N, BLOCKSIZE));
  reduceKernel<<<blockPerGrid, threadPerBlock>>>(A, out, N);
  cudaDeviceSynchronize();
}
```



## FlashAttention（PyTorch 版）

- $Q K^\top / \sqrt{d}$ → S → softmax → P → @V → (divide) → O
- Q, K, V: [N, d] → S / P: [N, N]

```python
def flash_attention_forward(Q, K, V, Br, Bc):
  N, d = Q.shape
  scale = 1.0 / math.sqrt(d)
  O = torch.zeros(N, d)
  L = torch.zeros(N)         # logsumexp, 留给 backward
  Tr = (N + Br - 1) // Br    # tiles of row
  Tc = (N + Bc - 1) // Bc    # tiles of col

  # loop Q
  for i in range(Tr):
    Qi = Q[i*Br : (i+1)*Br]                 # q - tile
    Oi = torch.zeros(Br, d)                 # out - this line
    mi = torch.full((Br,), float('-inf'))   # softmax - m
    li = torch.zeros(Br)                    # softmax - l
    # loop K / V
    for j in range(Tc):
      # slice K / V
      Kj = K[j*Bc : (j+1)*Bc]
      Vj = V[j*Bc : (j+1)*Bc]
      # Sij → Pij → O
      Sij = Qi @ Kj.transpose(-1, -2) * scale
      # compute this tile
      m_blk = Sij.max(dim=-1).values        # softmax - block max
      m_new = torch.maximum(mi, m_blk)      # softmax - row max
      Pij = torch.exp(Sij - m_new[:, None])
      alpha = torch.exp(mi - m_new)         # softmax - adjust scale
      # update this tile
      li = alpha * li + Pij.sum(dim=-1)
      Oi = Oi * alpha[:, None] + Pij @ Vj
      mi = m_new                            # 关键：提交新 max
    # update O result
    O[i*Br : (i+1)*Br] = Oi / li[:, None]
    L[i*Br : (i+1)*Br] = mi + torch.log(li)

  return O, L
```
